Pipeline Template

General preprocessor and model training pipeline boilerplate code:

from sklearn.compose import ColumnTransformer
from sklearn.pipeline import Pipeline
from sklearn.preprocessing import OneHotEncoder, StandardScaler

def build_pipeline(X, model):
    cat_cols = X.select_dtypes(include=["object"]).columns.tolist()
    num_cols = X.select_dtypes(include=[np.number]).columns.tolist()

    preprocessor = ColumnTransformer(transformers=[
        ("num", StandardScaler(), num_cols),
        ("cat", OneHotEncoder(handle_unknown="ignore", drop="first"), cat_cols)
    ])

    return Pipeline([
        ("preprocess", preprocessor),
        ("model", model)
    ])

GroupKFold Cross-Validation Template

Clean code to validate models across unseen regional groups:

from sklearn.model_selection import GroupKFold
from sklearn.base import clone

def evaluate_group_cv(X, y, groups, pipeline, n_splits=5):
    cv = GroupKFold(n_splits=n_splits)
    scores = []

    for train_idx, test_idx in cv.split(X, y, groups):
        model = clone(pipeline)
        model.fit(X.iloc[train_idx], y.iloc[train_idx])
        preds = model.predict(X.iloc[test_idx])
        scores.append(evaluate_metrics(y.iloc[test_idx], preds))
    return scores
